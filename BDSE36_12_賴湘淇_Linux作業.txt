#!/bin/bash

for name in $(cat $HOME/LinuxHW/test_names.txt);
do
        sudo useradd "$name" -m -s /bin/bash
        sudo usermod -aG airflow "$name"
        if [[ "$name" == "Alice" || "$name" == " Benny" ]];
        then
                sudo usermod -aG sudo "$name"
        fi
done
